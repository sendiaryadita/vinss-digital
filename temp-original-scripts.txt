function getTemplateText(id){
  const t = document.getElementById(id);
  if(!t) {
    console.warn('Template not found:', id);
    return '';
  }
  
  // For <template> elements, use .content.textContent or .innerHTML
  let text = '';
  if(t.tagName === 'TEMPLATE') {
    text = t.innerHTML.trim();
  } else {
    text = t.textContent.trim();
  }
  
  console.log('Template:', id, 'Length:', text.length);
  return text;
}

function showToast(msg){
  const el = document.getElementById('toast');
  if(!el) return;
  el.textContent = msg;
  el.classList.remove('opacity-0');
  el.classList.add('opacity-100');
  setTimeout(()=>{ el.classList.add('opacity-0'); el.classList.remove('opacity-100'); }, 1600);
}

  const href = 'https://wa.me/' + phone + '?text=' + encodeURIComponent(msg);

  if(btn && phone) btn.setAttribute('href', href);
}

function orderViaWA(templateId){
  console.log('=== Order VIA WA START ===');
  console.log('templateId:', templateId);
  
  if(!WA_NUMBER){
    console.error('WA_NUMBER not configured');
    showToast('Nomor WhatsApp belum dikonfigurasi');
    return;
  }
  console.log('WA_NUMBER:', WA_NUMBER);
  
  const txt   = getTemplateText(templateId);
  console.log('Template text retrieved, length:', txt.length);
  
  if(!txt){
    console.error('Template text is empty');
    showToast('Format template tidak ditemukan');
    return;
  }
  
  const phone = WA_NUMBER.replace(/\D/g,'');
  console.log('Phone after cleanup:', phone);
  
  const ok    = /^[1-9]\d{7,15}$/.test(phone);
  console.log('Phone validation:', ok);
  
  if(!ok){ 
    console.error('Phone validation failed');
    showToast('Nomor WA belum valid'); 
    return; 
  }

  const isMobile = /Android|iPhone|iPad|iPod|Windows Phone/i.test(navigator.userAgent);
  console.log('Is mobile:', isMobile);
  
  const waWeb = 'https://wa.me/' + phone + '?text=' + encodeURIComponent(txt);
  const waApp = 'whatsapp://send?phone=' + phone + '&text=' + encodeURIComponent(txt);
  
  console.log('waWeb URL:', waWeb.substring(0, 80) + '...');
  console.log('waApp URL:', waApp.substring(0, 80) + '...');

  const href = isMobile ? waApp : waWeb;
  console.log('Using href:', href.substring(0, 80) + '...');
  
  try {
    console.log('Attempting window.open...');
    const win  = window.open(href, '_blank');
    if(!win){ 
      console.warn('Window.open returned null, using fallback');
      window.location.href = waWeb; 
    } else {
      console.log('Window opened successfully');
    }
  } catch(e) {
    console.error('Exception in window.open:', e.message);
    console.log('Using fallback');
    window.location.href = waWeb;
  }
  console.log('=== Order VIA WA END ===');
}

// ====== BIND COPY & ORDER BUTTONS ======
function bindCopyAndOrder(){
  // Copy Format
  const copyBtns = document.querySelectorAll('.copyBtn');
  copyBtns.forEach(btn=>{
    btn.addEventListener('click', async ()=>{
      const id  = btn.getAttribute('data-copy');
      const txt = getTemplateText(id);
      if(!txt){
        showToast('Template tidak ditemukan');
        return;
      }
      try {
        await navigator.clipboard.writeText(txt);
        showToast('Format disalin ke clipboard');
      } catch(e){
        showToast('Gagal menyalin');
      }
    });
  });

  // Order via WhatsApp
  const orderBtns = document.querySelectorAll('.orderBtn');
  orderBtns.forEach(btn=>{
    btn.addEventListener('click', ()=>{
      const id = btn.getAttribute('data-order');
      console.log('Order button clicked, templateId:', id);
      try {
        orderViaWA(id);
      } catch(e) {
        console.error('Order error:', e);
        showToast('Terjadi error saat order: ' + e.message);
      }
    });
  });
}

// ====== BIND SEARCH ======
function bindSearch(){
  const searchInput = document.getElementById('searchInput');
  const serviceGrid = document.getElementById('serviceGrid');
  if (searchInput && serviceGrid){
    const cards = Array.from(serviceGrid.children);
    searchInput.addEventListener('input', () => {
      const q = searchInput.value.trim().toLowerCase();
      cards.forEach(card => {
        const name = card.innerText.trim().toLowerCase();
        card.style.display = name.includes(q) ? '' : 'none';
      });
    });
  }
}

// ====== INIT ======

  bindNavigationButtons();
  bindCopyAndOrder();
  bindSearch();
  setWALink();
}

// Jalankan saat DOM siap
if(document.readyState === 'loading'){
  document.addEventListener('DOMContentLoaded', initApp);
} else {
  initApp();
}
